78 /**
79  * \brief Wait until all necessary cores reached the synchronization barrier, stripped version of EE_hal_sync_barrier
80  * \return None
81  */
82 void __attribute__ ((noinline)) ILM_OS_SyncCores(void)
83 {
84 #if ILM_USE_OS_MIN_BARRIER > 0
85  EE_hal_spin_in(ILM_OS_SPINLOCK);
86  /* Set current CPU as entered */
87  _ILM_OS_Barrier.value |= ((EE_UINT32)1U << EE_CURRENTCPU);
88  EE_hal_spin_out(ILM_OS_SPINLOCK);
89  /* Wait for all other cores/CPUs */
90  while ( _ILM_OS_Barrier.value != _ILM_OS_BarrierMask );
91 #else
92  EE_hal_sync_barrier(&_ILM_OS_Barrier, &_ILM_OS_BarrierMask);
93 #endif
94 }
