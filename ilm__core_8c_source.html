62 /**
63  * \brief Internal function for read pointer backup
64  * \param pData Pointer to data struct
65  * \return None
66  */
67 static void _ILM_BackupReadPointer(ILM_DATA_IN_T *pData)
68 {
69  /*
70  * Backup Input Data
71  * "Zero-Time" Communication
72  * -> Enforce "Delayed Read"
73  */
74  if(pData)
75  {
76  while(pData->pInput)
77  {
78  pData->pLocalCopy = pData->pInput->Read.Generic;
79  pData++;
80  }
81  }
82 }

84 /**
85  * \brief Internal function for event table processing
86  * \param pEntry Current table entry
87  * \return None
88  */
89 static void _ILM_EventHandling(ILM_LET_TABLE_T * pEntry)
90 {
91  uint16 OldOffset = pEntry->Offset;
92  /*
93  * Process all entries with the same offset
94  */
95  while(pEntry->Offset == OldOffset )
96  {
97  if((pEntry->Type & ILM_EVENT_TACT) == ILM_EVENT_TACT)
98  {
99  /*
100  * Backup read pointer first if needed
101  */
102  _ILM_BackupReadPointer(pEntry->Payload.pDataIn);
103  /*
104  * Activate task
105  */
106  ILM_OS_ActivateTask(pEntry->TaskId);
107 #if ILM_USE_HOOK_TACT > 0
108  ILM_HOOK_TACT(pEntry);
109 #endif
110  }
111  else
112  {
113 #if ILM_USE_PRIOB > 0
114  if((pEntry->Type & ILM_EVENT_PRIOB) == ILM_EVENT_PRIOB)
115  {
116  /* not implmented yet*/
117 #if ILM_USE_HOOK_PRIOB > 0
118  ILM_HOOK_PRIOB(pEntry);
119 #endif
120  }
121 #endif
122  if((pEntry->Type & ILM_EVENT_PSWAP) == ILM_EVENT_PSWAP)
123  {
124  /*
125  * Simple pointer swap
126  * "Zero-Time" Communication
127  * -> Enforce "Early Write"
128  * -> Marks end of LET
129  */
130 #if ILM_USE_HOOK_MISS_LET > 0
131  if(ILM_OS_CheckTaskState(pEntry->TaskId) == 1)
132  ILM_HOOK_MISS_LET(pEntry);
133 #endif
134  ILM_DATA_OUT_T * pData = pEntry->Payload.pDataOut;
135  void * Backup = pData->Read.Generic;
136  pData->Read.Generic = pData->Write.Generic;
137  pData->Write.Generic = Backup;
138 #if ILM_USE_HOOK_PSWAP > 0
139  ILM_HOOK_PSWAP(pEntry);
140 #endif
141  }
142  if((pEntry->Type & ILM_EVENT_LETEV) == ILM_EVENT_LETEV)
143  {
144  /*
145  * Backup read pointer first if needed
146  */
147  _ILM_BackupReadPointer(pEntry->Payload.pDataIn);
148 #if ILM_USE_OSEK_ECC > 0
149  /*
150  * Use set event if ECC tasks are used
151  */
152  ILM_OS_SetEvent(pEntry->TaskId, pEntry->Payload.pSchedEvent);
153 #else
154  /*
155  * Use task activate if BCC tasks are used
156  */
157  ILM_OS_ActivateTask(pEntry->TaskId);
158 #endif
159 #if ILM_USE_HOOK_LETEV > 0
160  ILM_HOOK_LETEV(pEntry);
161 #endif
162  }
163  if((pEntry->Type & ILM_EVENT_TTERM) == ILM_EVENT_TTERM)
164  {
165 #if ILM_USE_TERM_CTX > 0
166  /*
167  * Set event for task termination
168  */
169  ILM_OS_SetEvent(pEntry->TaskId, pEntry->Payload.pSchedEvent);
170 #endif
171 #if ILM_USE_HOOK_TTERM > 0
172  ILM_HOOK_TTERM(pEntry);
173 #endif
174  }
175  }
176  _ILM_TableIdx = (_ILM_TableIdx + 1) % _ILM_TableSize;
177  OldOffset = pEntry->Offset;
178  pEntry = &_ILM_Table[_ILM_TableIdx];
179 
180  }
181 
182 }

184 /**
185  * \brief Handler function for IRQ or time tick based LET event processing\n
186  * If ILM_USE_HW_CCU == 0 the function should be called within the time tick\n
187  * for software based CCU emulation
188  * \param CurTick Current time tick value, only needed if ILM_Hander is called\n
189  * from a task context
190  * \return None
191  */
192 void __attribute__((unused)) ILM_Handler(uint32 CurTick)
193 {
194 #if ILM_USE_HOOK_START_ILM > 0
195  ILM_HOOK_START_ILM_HANDLER();
196 #endif
197  if(!_ILM_Table)
198  return;
199 #if ILM_USE_HW_CCU > 0
200  (void)CurTick;
201 #endif
202 #if ILM_USE_HW_CCU == 0
203  if(_ILM_SystemReady > 0)
204  {
205  if(_ILM_SystemReady == 1)
206  {
207  _ILM_LastStart = CurTick;
208  _ILM_SystemReady = 2;
209  }
210 #endif
211  ILM_LET_TABLE_T * pEntry = &_ILM_Table[_ILM_TableIdx];
212 #if ILM_USE_HW_CCU == 0
213  if(_ILM_LastStart + pEntry->Offset == CurTick )
214  {
215 #endif
216  /*
217  * Call LET event handler
218  */
219  _ILM_EventHandling(pEntry);
220 #if ILM_USE_HW_CCU == 0
221  if(_ILM_TableIdx == 0)
222  _ILM_LastStart = _ILM_LastStart + _ILM_MacroPeriod;
223  }
224  }
225 #endif
226 #if ILM_USE_HOOK_STOP_ILM > 0
227  ILM_HOOK_STOP_ILM_HANDLER();
228 #endif
229 }

232 /**
233  * \brief Init function for ILM core module
234  * \param pTable Pointer to LET table
235  * \param TableSize Length of the table located at pTable
236  * \param MacroPeriod Macro period of the provided LET table
237  * \return None
238  */
239 void __attribute__((unused)) ILM_Init(ILM_LET_TABLE_T *pTable, uint32 TableSize, uint32 MacroPeriod)
240 {
241  if(pTable == (ILM_LET_TABLE_T *)0)
242  return;
243  if(TableSize == 0)
244  return;
245  if(MacroPeriod == 0)
246  return;
247 
248  _ILM_SystemReady = 0;
249  _ILM_Table = pTable;
250  _ILM_TableSize = TableSize;
251  _ILM_MacroPeriod = MacroPeriod;
252  _ILM_TableIdx = 0;
253  _ILM_SystemReady = 1;
254 }
