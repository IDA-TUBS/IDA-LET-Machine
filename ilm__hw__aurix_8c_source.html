94 /**
95  * \brief Init function for CCU based IRQ generation
96  * \param pTable Pointer to compare table
97  * \param TableSize Length of referenced compare table
98  * \param Freq Tick frequency(granularity) for compare table entries
99  * \param MacroPeriod Macro period of compare table
100  * \return None
101  */
102 void __attribute__((unused)) ILM_HW_Init(ILM_HW_LET_TABLE_T * pTable, uint32 TableSize, uint32 Freq, uint32 MacroPeriod)
103 {
104 
105  if(pTable == NULL || Freq == 0 || MacroPeriod == 0)
106  return;
107 
108  _ILM_HW_Table = pTable;
109  _ILM_HW_TableIdx = 0;
110  _ILM_HW_TableSize = TableSize;
111 
112  /*
113  * Initialize configuration struct
114  */
115  IfxCcu6_Timer_Config TimerConfig;
116  IfxCcu6_Timer_initModuleConfig(&TimerConfig, &MODULE_CCU60);
117 
118  /*
119  * Set "PWM" frequency to (Freq/MacroPeriod)
120  * (Ticks/s) / (Ticks / Period)
121  *
122  */
123  TimerConfig.base.t13Frequency = ((float32)Freq)/((float32)MacroPeriod);
124  TimerConfig.base.t13Period = MacroPeriod;
125 
126  /*
127  * Use T13, without synchronousOperation
128  */
129  TimerConfig.timer = IfxCcu6_TimerId_t13;
130  TimerConfig.synchronousOperation = FALSE;
131 
132  /*
133  * Setup clock source, reset counter value
134  */
135  TimerConfig.clock.t13ExtClockEnabled = FALSE;
136  TimerConfig.clock.t13ExtClockInput = NULL_PTR;
137  TimerConfig.clock.t13countingInputMode = IfxCcu6_CountingInputMode_internal;
138  TimerConfig.timer13.counterValue = 0;
139 
140  /*
141  * Setup CCU IRQ
142  */
143  TimerConfig.interrupt1.source = IfxCcu6_InterruptSource_t13CompareMatch;
144  TimerConfig.interrupt1.serviceRequest = IfxCcu6_ServiceRequest_1;
145  TimerConfig.interrupt1.priority = ILM_HW_HARD_IRQ;
146  TimerConfig.interrupt1.typeOfService = (IfxSrc_Tos)IfxCpu_getCoreId();
147 
148  /*
149  * Neither input, nor output triggering
150  */
151  TimerConfig.pins = NULL_PTR;
152  TimerConfig.trigger.t12ExtInputTrigger = NULL_PTR;
153  TimerConfig.trigger.t13ExtInputTrigger = NULL_PTR;
154  TimerConfig.trigger.extInputTriggerMode = IfxCcu6_ExternalTriggerMode_disable;
155  TimerConfig.trigger.t13InSyncWithT12 = FALSE;
156 
157  /*
158  * Pass config struct to init function
159  */
160  IfxCcu6_Timer_initModule((IfxCcu6_Timer *)&_ILM_HW_TIMER, &TimerConfig);
161 
162  /*
163  * Calculate scaling factor based on the internal
164  * clock source.
165  */
166  _ILM_HW_PeriodScale =((float32)_ILM_HW_TIMER.ccu6->T13PR.U) / ((float32)MacroPeriod);
167 
168 #if ILM_USE_HW_SOFT_IRQ > 0
169  /*
170  * Setup core-2-core software IRQs
171  */
172  SRC_GPSR11.U = ILM_TC2YX_SRN_TYPE_OF_SERVICE(1U) | ILM_TC2YX_SRN_ENABLE | ILM_TC2YX_SRN_PRIORITY(ILM_HW_SOFT_IRQ);
173  SRC_GPSR21.U = ILM_TC2YX_SRN_TYPE_OF_SERVICE(2U) | ILM_TC2YX_SRN_ENABLE | ILM_TC2YX_SRN_PRIORITY(ILM_HW_SOFT_IRQ);
174 #endif
175 
176 #if defined(ERIKA_OS)
177  /*
178  * Not needed, as done by ERIKA
179  */
180 #else
181  /*
182  * Register IRQ handler
183  */
184  IfxCpu_Irq_installInterruptHandler(&ILM_HW_Isr_Master, ILM_HW_HARD_IRQ);
185  IfxCpu_enableInterrupts();
186 #endif
187 
188 }

202 /**
203  * \brief IRQ handler for master core C0, called from OS implementation
204  * \return State for local ILM_Handler (1 execute, 0 skip)
205  */
206 uint32 __attribute__((unused)) ILM_HW_Isr_Master(void)
207 {
208  uint32 ret = 0;
209  uint32 CoreMask = _ILM_HW_Table[_ILM_HW_TableIdx].CoreMask;
210  uint16 NextIdx = (_ILM_HW_TableIdx + 1) % _ILM_HW_TableSize;
211 
212  /*
213  * Handle CCU IRQ,
214  * set core sync barrier,
215  * set new reload value and
216  * restart the timer
217  */
218  IfxCcu6_Timer_stop((IfxCcu6_Timer *)&_ILM_HW_TIMER);
219  ILM_OS_Set_BarrierMask(CoreMask);
220  IfxCcu6_setT13CaptureCompareState(_ILM_HW_TIMER.ccu6, IfxCcu6_CaptureCompareState_clear);
221  _ILM_HW_TIMER.ccu6->CC63SR.B.CCS = _ILM_HW_CalcCompare(NextIdx);
222  IfxCcu6_Timer_start((IfxCcu6_Timer *)&_ILM_HW_TIMER);
223 
224  /*
225  * Redirect IRQ to other cores
226  */
227 #if ILM_USE_HW_SOFT_IRQ == 0
228  ILM_OS_ActivateILMCoreTasks(CoreMask);
229 #else
230  uint32 i = 0;
231  for(i=1; i < ILM_NUM_OF_CORES; i++)
232  {
233  if(CoreMask & (1 << i))
234  _ILM_HW_SendSoftIrq(i);
235  }
236  /*
237  * Check if this core should also
238  * call the ILM_Handler
239  */
240  ret = CoreMask & (ILM_COREMASK_C0);
241 #endif
242  _ILM_HW_TableIdx = NextIdx;
243 
244  return ret;
245 }
